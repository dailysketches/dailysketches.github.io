		float increment;
ofxTexturePlane plane;

void ofApp::setup(){
    increment = 0.02;
    plane.setup("partially-transparent.png");
}

void ofApp::update(){
    plane.incrementTextureScale(-increment);
}

void ofApp::draw(){
    ofBackground(ofColor::red);
    plane.draw();
}
