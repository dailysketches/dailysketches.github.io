}

void ofApp::drawAnim() {
    ofBackground(0, 0, 0);
    
    for(int i = 0; i &lt; num; i++) {
        ofSetColor(255, 255, 255, 100);
        if(i % 2 == 0) ofSetColor(255, 0, 0, 100);
        ofRect(ofRandom(-max, width),
               ofRandom(-max, height),
               ofRandom(max),
               ofRandom(max));
    }
    
    if(ofGetFrameNum() == 150) {
        renderGif();
    }
